# Foundation F1 — Tenanted Skeleton Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Stand up a multi-tenant API skeleton that resolves tenants from subdomains, routes each request to that tenant's own Postgres database, and proves data isolation with integration tests against real Postgres.

**Architecture:** pnpm + Turborepo monorepo with a Fastify API, Drizzle ORM, and a tenant-aware request pipeline. A control-plane Postgres database stores the tenant registry; each tenant gets its own database. A `tenant:create` CLI provisions tenants. OpenAPI is generated from Zod schemas via `fastify-zod-openapi`. Observability via Pino + Sentry.

**Tech Stack:** TypeScript 5.x (strict, ESM), Node 20+, pnpm 9, Turborepo, Fastify 5, Zod 3, `fastify-zod-openapi`, Drizzle ORM, `drizzle-kit`, `pg`, `dotenv`, `pino`, `pino-pretty`, `@sentry/node`, Vitest, `@testcontainers/postgresql`, `tsx`.

**What this plan does NOT include** (covered by later plans):
- F2: users / sessions / login / MFA / RBAC / audit log / transactional email
- F3: module SDK / static plugin registry / connector framework / pg-boss / demo module
- Web frontend, deployment config

```ts
import pg from 'pg';
import * as schema from './schema.js';

export type ControlPlaneDb = NodePgDatabase<typeof schema>;

let cached: { pool: pg.Pool; db: ControlPlaneDb } | null = null;

export function getControlPlaneDb(url: string): ControlPlaneDb {
  if (cached) return cached.db;
  const pool = new pg.Pool({ connectionString: url, max: 10 });
  const db = drizzle(pool, { schema });
  cached = { pool, db };
  return db;
}

export async function closeControlPlaneDb(): Promise<void> {
  if (cached) {
    await cached.pool.end();
    cached = null;
  }
}
```

```ts
import pg from 'pg';
import path from 'node:path';
import { fileURLToPath } from 'node:url';

const here = path.dirname(fileURLToPath(import.meta.url));

const CONTROL_MIGRATIONS = path.resolve(here, '../migrations/control-plane');
const TENANT_MIGRATIONS = path.resolve(here, '../migrations/tenant');

export async function migrateControlPlane(url: string): Promise<void> {
  const pool = new pg.Pool({ connectionString: url, max: 1 });
  try {
    const db = drizzle(pool);
    await migrate(db, { migrationsFolder: CONTROL_MIGRATIONS });
  } finally {
    await pool.end();
  }
}

export async function migrateTenant(url: string): Promise<void> {

  try {
    const db = drizzle(pool);
```

```ts
import { drizzle, type NodePgDatabase } from 'drizzle-orm/node-postgres';
import pg from 'pg';
import * as schema from './schema.js';

export type TenantDb = NodePgDatabase<typeof schema>;

export interface TenantConnectionInfo {
  host: string;
  dbName: string;
  user: string;
  password: string;
}

export interface TenantConnection {
  db: TenantDb;
  close(): Promise<void>;
}

export function buildTenantUrl(info: TenantConnectionInfo, template: string): string {
  if (template.includes('{{db_name}}')) {
    return template.replace('{{db_name}}', info.dbName);
  }
  return `postgres://${info.user}:${info.password}@${info.host}/${info.dbName}`;
}

export function createTenantConnection(url: string): TenantConnection {
  const pool = new pg.Pool({ connectionString: url, max: 5, idleTimeoutMillis: 30_000 });
  const db = drizzle(pool, { schema });
  return {
    db,
    async close() {
      await pool.end();
    },
  };
}
```

```ts
#!/usr/bin/env tsx
import 'dotenv/config';
import { createTenant } from './create.js';
import { migrateAllTenants, migrateOneTenant } from './migrate.js';

const [, , cmd, ...rest] = process.argv;

function parseFlags(args: string[]): Record<string, string> {
  const out: Record<string, string> = {};
  for (let i = 0; i < args.length; i++) {
    const a = args[i];
    if (a?.startsWith('--')) {
      const key = a.slice(2);
      const val = args[i + 1];
      if (val === undefined || val.startsWith('--')) {
        out[key] = 'true';
      } else {
        out[key] = val;
        i++;
      }
    }
  }
  return out;
}

async function main() {
  const flags = parseFlags(rest);
  switch (cmd) {
    case 'create': {
      const slug = flags.slug;
      const name = flags.name;
      if (!slug || !name) {
        console.error('Usage: tenant:create --slug <slug> --name "<name>"');
        process.exit(1);
      }
      await createTenant({ slug, name });
      break;
    }
    case 'migrate': {
      const slug = flags.slug;
      if (!slug) {
        console.error('Usage: tenant:migrate --slug <slug>');
        process.exit(1);
      }
      await migrateOneTenant(slug);
      break;
    }
    case 'migrate-all': {
      await migrateAllTenants();
      break;
    }
    default:
      console.error('Unknown command. Use: create | migrate | migrate-all');
      process.exit(1);
  }
}

main().catch((err) => {
  console.error(err);
  process.exit(1);
});
```

```ts
const SLUG_RE = /^[a-z][a-z0-9-]{1,30}[a-z0-9]$/;

export async function createTenant(opts: { slug: string; name: string }): Promise<void> {
  const { slug, name } = opts;

  if (!SLUG_RE.test(slug)) {
    throw new Error(
      `Invalid slug "${slug}". Must be lowercase, 3-32 chars, [a-z0-9-], start with a letter, end alphanumeric.`
    );
  }

  const controlUrl = required('CONTROL_PLANE_DATABASE_URL');
  const adminUrl = required('POSTGRES_ADMIN_URL');
  const template = required('TENANT_DATABASE_URL_TEMPLATE');

  // Ensure control plane is migrated before doing anything else
  await migrateControlPlane(controlUrl);

  const dbName = `tenant_${slug.replace(/-/g, '_')}`;
  const tenantUrl = template.replace('{{db_name}}', dbName);

  const control = getControlPlaneDb(controlUrl);
  const existing = await control.select().from(tenants).where(eq(tenants.slug, slug));
  if (existing.length > 0) {
    throw new Error(`Tenant with slug "${slug}" already exists`);
  }

  // 1. Create the tenant database (via admin connection)
  const admin = new pg.Client({ connectionString: adminUrl });
  await admin.connect();
  try {
    // CREATE DATABASE cannot be parameterized; we already validated `dbName` is safe
    // because it derives from a regex-validated slug.
    await admin.query(`CREATE DATABASE "${dbName}"`);
  } finally {
    await admin.end();
  }

  // 2. Apply tenant migrations to the new DB
  await migrateTenant(tenantUrl);

  // 3. Parse template back into host/user/password parts for the tenants row
  const parsed = new URL(tenantUrl);
  const host = parsed.host;
  const user = decodeURIComponent(parsed.username);
  const password = decodeURIComponent(parsed.password);

  // 4. Insert into control-plane tenants table
  const [row] = await control
    .insert(tenants)
    .values({
      slug,
      name,
      dbHost: host,
      dbName,
      dbUser: user,
      dbPassword: password,
    })
    .returning();

  await closeControlPlaneDb();

  console.log(`Created tenant "${slug}" (${row?.id}). DB: ${dbName}`);
}

function required(name: string): string {
  const v = process.env[name];
  if (!v) throw new Error(`${name} is not set`);
  return v;
```

```ts
export interface TenantRecord {
  id: string;
  slug: string;
  name: string;
  dbHost: string;
  dbName: string;
  dbUser: string;
  dbPassword: string;
}

export interface TenantCacheOptions<T> {
  ttlMs: number;
  fetcher: (slug: string) => Promise<T | null>;
  now?: () => number;
}

interface Entry<T> {
  value: T | null;
  expiresAt: number;
}

export class TenantCache<T = TenantRecord> {
  private readonly ttlMs: number;
  private readonly fetcher: (slug: string) => Promise<T | null>;
  private readonly now: () => number;
  private readonly store = new Map<string, Entry<T>>();

  constructor(opts: TenantCacheOptions<T>) {
    this.ttlMs = opts.ttlMs;
    this.fetcher = opts.fetcher;
    this.now = opts.now ?? (() => Date.now());
  }

  async get(slug: string): Promise<T | null> {
    const t = this.now();
    const hit = this.store.get(slug);
    if (hit && hit.expiresAt > t) return hit.value;
    const value = await this.fetcher(slug);
    this.store.set(slug, { value, expiresAt: t + this.ttlMs });
    return value;
  }

  invalidate(slug: string): void {
    this.store.delete(slug);
  }

  clear(): void {
    this.store.clear();
  }
}
```

```ts
import { TenantCache, type TenantRecord } from '../tenant-cache.js';

declare module 'fastify' {
  interface FastifyRequest {
    tenant?: TenantRecord;
  }
}

export interface TenantResolverOptions {
  baseDomain: string; // e.g. "businessos.localhost"
  cache: TenantCache;
  /**
   * If true, requests whose Host is exactly the base domain (no subdomain)
   * are allowed through with req.tenant left undefined.
   * Use this for control-plane / health routes.
   */
  allowControlPlane?: boolean;
}

function extractSubdomain(host: string, baseDomain: string): string | null {
  // Strip port if present
  const justHost = host.split(':')[0]!;
  if (justHost === baseDomain) return null;
  if (!justHost.endsWith('.' + baseDomain)) return null;
  return justHost.slice(0, -1 - baseDomain.length);
}

const plugin: FastifyPluginAsync<TenantResolverOptions> = async (
  app: FastifyInstance,
  opts: TenantResolverOptions
) => {
  app.addHook('onRequest', async (req: FastifyRequest, reply) => {
    const host = req.headers.host;
    if (!host) {
      return reply.code(400).send({ error: 'Missing Host header' });
    }

    const slug = extractSubdomain(host, opts.baseDomain);

    if (slug === null) {
      if (opts.allowControlPlane) return;
      return reply.code(404).send({ error: 'Tenant not found' });
    }

    const tenant = await opts.cache.get(slug);
    if (!tenant) {
      return reply.code(404).send({ error: 'Tenant not found' });
    }

    req.tenant = tenant;
    req.log.info({ tenant_slug: tenant.slug, tenant_id: tenant.id }, 'tenant resolved');
  });
};

export const tenantResolver = fp(plugin, { name: 'tenant-resolver' });

// Exported for unit testing
export const __testing__ = { extractSubdomain };
```

## Task 12: Tenant DB binder plugin + per-tenant pool LRU

**Files:**

`TenantPoolCache` keeps a bounded LRU of per-tenant Drizzle instances. `db-binder` plugin acquires one for the resolved tenant and attaches it as `req.db`.

- [ ] **Step 1: Create `apps/api/src/tenant-pool-cache.ts`**

```ts
import { createTenantConnection, type TenantConnection } from '@business-os/db/tenant';

export interface TenantPoolCacheOptions {
  urlTemplate: string;
  max: number;
}

interface Entry {
  conn: TenantConnection;
  lastUsedAt: number;
}

export class TenantPoolCache {
  private readonly urlTemplate: string;
  private readonly max: number;
  private readonly entries = new Map<string, Entry>();

  constructor(opts: TenantPoolCacheOptions) {
    this.urlTemplate = opts.urlTemplate;
    this.max = opts.max;
  }

  acquire(dbName: string): TenantConnection {
    const now = Date.now();
    const hit = this.entries.get(dbName);
    if (hit) {
      hit.lastUsedAt = now;
      return hit.conn;
    }
    const url = this.urlTemplate.replace('{{db_name}}', dbName);
    const conn = createTenantConnection(url);
    this.entries.set(dbName, { conn, lastUsedAt: now });
    this.evictIfNeeded();
    return conn;
  }

  private evictIfNeeded(): void {
    while (this.entries.size > this.max) {
      let oldestKey: string | null = null;
      let oldestAt = Infinity;
      for (const [k, e] of this.entries) {
        if (e.lastUsedAt < oldestAt) {
          oldestAt = e.lastUsedAt;
          oldestKey = k;
        }
      }
      if (!oldestKey) break;
      const dropped = this.entries.get(oldestKey)!;
      this.entries.delete(oldestKey);
      // Fire and forget; pool will close gracefully
      dropped.conn.close().catch(() => undefined);
    }
  }

  async closeAll(): Promise<void> {
    const all = [...this.entries.values()];
    this.entries.clear();
    await Promise.all(all.map((e) => e.conn.close()));
  }

  size(): number {
    return this.entries.size;
  }
}
```

```ts
      const row = rows[0];
      if (!row) return null;
      if (row.status !== 'active') return null;
      return {
        id: row.id,
        slug: row.slug,
        name: row.name,
        dbHost: row.dbHost,
        dbName: row.dbName,
        dbUser: row.dbUser,
        dbPassword: row.dbPassword,
      };
    },
  });

  const pools = new TenantPoolCache({
    urlTemplate: env.TENANT_DATABASE_URL_TEMPLATE,
    max: 100,
  });

  const app = Fastify({
    loggerInstance: logger,
    disableRequestLogging: false,
    trustProxy: true,
  });

  app.setErrorHandler((err, req, reply) => {
    req.log.error({ err }, 'request failed');
    captureException(err, {
      tenant_slug: req.tenant?.slug ?? '',
      route: req.routeOptions?.url ?? '',
    });
    reply.code(500).send({ error: 'Internal Server Error' });
  });

  await app.register(tenantResolver, {
    baseDomain: env.API_BASE_DOMAIN,
    cache,
    allowControlPlane: true,
  });
  await app.register(dbBinder, { pools });
  await app.register(healthRoutes);

  return { app, pools, cache };
}
```

## Notes for the executor

- **Frequent commits are the point.** Don't squash; the history is the record of what changed when.
- If a step fails, STOP. Don't paper over with try/catch. Read the error, understand why, fix root cause. Use systematic-debugging if needed.

- Subdomain `<tenant>.businessos.localhost` resolves to `127.0.0.1` on most modern OSes without /etc/hosts changes (this is a public-suffix-list shortcut for `.localhost`). If your machine does not, add `127.0.0.1 demo.businessos.localhost` to your hosts file or use `curl --resolve`.
